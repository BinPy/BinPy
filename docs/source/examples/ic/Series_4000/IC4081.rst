    # The Pin configuration is:
    
    inp = {1: 0, 2: 0, 5: 0, 6: 1, 7: 0, 8: 1, 9: 0, 12: 1, 13: 1, 14: 1}
    
    # Pin initinalization
    
    # Powering up the IC - using -- ic.set_IC({14: 1, 7: 0}) -- \n
    
    ic.set_IC({14: 1, 7: 0})
    
    # Setting the inputs of the ic
    
    ic.set_IC(inp)
    
    # Draw the IC with the current configuration\n
    
    ic.draw_IC()
